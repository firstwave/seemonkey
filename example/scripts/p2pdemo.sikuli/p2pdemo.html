import unittest
import sys

from com.criticalpath.seemonkey import SeeMonkey
from com.android.monkeyrunner import MonkeyDevice
from org.sikuli.script import Settings

Settings.MoveMouseDelay = 0.01
Settings.AutoWaitTimeout = 10

Settings.ActionLogs = False
Settings.DebugLogs = False
print "set up android connection"
scr = SeeMonkey() # compatiable with Sikuli Screen/Region

assert scr != None
dev = scr.getMonkeyDevice() # Android Monkey device
scr.autoDelay = 250



class TestAndroidBasic(unittest.TestCase):
    def testA_setup(self):
        scr.wake()
        if scr.exists():
            if scr.exists():
                scr.dragDrop(Pattern().similar(0.73),)
            else:
                scr.dragDrop(Pattern().similar(0.73),)
        while not scr.exists():
            scr.press('HOME')

        # Clear paypal data
        scr.press('MENU')
        scr.click()
        scr.click()
        scr.click()
        while not scr.exists():
            scr.press('DPAD_DOWN')
            scr.press('DPAD_DOWN')
            scr.press('DPAD_DOWN')

        scr.click()
        try:
            scr.click(Pattern().similar(0.92))
            scr.click(Pattern().similar(0.83))
            print("Data cleared.")
        except:
            # because the data may already be cleared
            pass

        # go back to homescreen and launch the paypal app

        while not scr.exists():
            scr.press('HOME')
        scr.click()
        while not scr.exists(Pattern().similar(0.88)):
            scr.press('DPAD_DOWN')
            scr.press('DPAD_DOWN')
            scr.press('DPAD_DOWN')

        scr.click()
        scr.wait(Pattern().similar(0.89), 10)

    def testB_UserAgreement(self):
        assert scr.exists(Pattern().similar(0.87))
        assert scr.exists(Pattern().similar(0.89))

        # verify links in user agreement
        # these are visible by default on WVGA screens
        scr.click()
        scr.wait(, 30) # indicates that the website has loaded
        assert scr.exists() # has the correct page loaded?

        scr.press('BACK')
        scr.click(Pattern().similar(0.89))
        scr.wait()

    def testC_Login(self):
        assert scr.exists()
        assert scr.exists(Pattern().similar(0.77)) # should be focused and empty
        scr.longClick()
        scr.wait(, 10)
        try:
            scr.click(Pattern().similar(0.99).targetOffset(-87,3))
            scr.click(Pattern().targetOffset(-182,9))
        except:
            # must already be selected
            pass
        scr.press('BACK')
        scr.wait()
        scr.type('op-us-p12@paypal.com')
        scr.press('ENTER')
        scr.type('11111111')
        if scr.exists():
            scr.press('BACK')
        scr.click()
        sleep(5) # wait five seconds, to see if login failed

        if scr.exists():
            # login failed, try again
            scr.click()
            scr.type('11111111')
            if scr.exists():
                scr.press('BACK')
            scr.click()

        scr.wait(, 30)


    def testD_SendMoney(self):
        scr.click()
        scr.wait()
        scr.type('op-us-p8@paypal.com')
        sleep(1)
        scr.press('DPAD_DOWN')
        scr.type('.05')
        if scr.exists():
            scr.press('BACK')
        scr.click(Pattern().targetOffset(-192,-2))
        scr.click()
        scr.wait(, 10)
        scr.click()
        scr.type('SeeMonkey 12345')
        if scr.exists():
            scr.press('BACK')
        scr.click()
    def testE_History(self):
        scr.wait(, 30)
        scr.click()
        scr.wait(, 60) # pull-down arrow on History entry
        scr.click(Pattern().targetOffset(-10,62))
        scr.wait()
        scr.click()
        scr.wait()
        assert scr.exists('12345')
        scr.click()
        scr.press("BACK")
        scr.press("MENU")
        scr.click()

if __name__ == '__main__':
    suite = unittest.TestLoader().loadTestsFromTestCase(TestAndroidBasic)
    unittest.TextTestRunner(verbosity=2).run(suite)
    exit(0)
